  <h2>Invia Email con Gmail</h2>


  <label>Destinatario</label>

  <input type="email" id="to" placeholder="es. esempio@gmail.com" />


  <label>Oggetto</label>

  <input type="text" id="subject" placeholder="Oggetto" />


  <label>Messaggio</label>

  <textarea id="body" rows="6" placeholder="Scrivi il tuo messaggio..."></textarea>


  <button id="send">Invia Email</button>


  <p id="status" style="margin-top:12px;color:#666;"></p>


  <script>

    document.getElementById("send").onclick = () => {

      const to = encodeURIComponent(document.getElementById("to").value);

      const subject = encodeURIComponent(document.getElementById("subject").value);

      const body = encodeURIComponent(document.getElementById("body").value);


      if (!to) {

        document.getElementById("status").textContent = "Inserisci un destinatario.";

        return;

      }


      const gmailUrl = `https://mail.google.com/mail/?view=cm&fs=1&to=${to}&su=${subject}&body=${body}`;


      window.open(gmailUrl, "_blank");


      document.getElementById("status").textContent = "Apertura finestra Gmail...";

    };

  </script>
